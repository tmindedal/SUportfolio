-- Opretter en tabel til at gemme data om helt.
CREATE TABLE IF NOT EXISTS heroes (
    name TEXT PRIMARY KEY,
    health INTEGER,
    maxHp INTEGER,
    strength INTEGER,
    xp INTEGER,
    level INTEGER,
    gold INTEGER,
    weapon_name TEXT,
    weapon_damage INTEGER,
    weapon_strengthModifier INTEGER,
    weapon_durability INTEGER
);
